20json TaskManager::connect_task(nlohmann::json device_list) {
21 logger::log(LEVEL_TRACE, "Received \"{}\" task", TASK_TYPE_CONNECT);
22
23 json output = {
24 {WORD_RESULT_ID, RESULT_OK_ID},
25 {WORD_RESULT_MSG, RESULT_OK_MSG}
26 };
27 json device_results;
28
29 for (const auto &json_item : device_list.items()) {
30 const std::string &device = json_item.key();
31
32 if (device == DEVICE_EXT_GEN) {
33 device_results[json_item.key()] = device_set.connect(DEVICE_GEN, device, json_item.value());
34
35 if (!device_results[json_item.key()]) {
36 output[WORD_RESULT_ID] = EXT_GEN_NO_CONNECTION_ID;
37 output[WORD_RESULT_MSG] = EXT_GEN_NO_CONNECTION_MSG;
38 break;
39 }
40 } else if (device == DEVICE_RBD_UPKB || device == DEVICE_RBD_TESART || device == DEVICE_RBD_DEMO) {
41 device_results[json_item.key()] = device_set.connect(DEVICE_RBD, device, json_item.value());
42
43 if (!device_results[json_item.key()]) {
44 output[WORD_RESULT_ID] = RBD_NO_CONNECTION_ID;
45 output[WORD_RESULT_MSG] = RBD_NO_CONNECTION_MSG;
46 break;
47 }
48 } else {
49 device_results[json_item.key()] = device_set.connect(DEVICE_VNA, device, json_item.value());
50
51 if (!device_results[json_item.key()]) {
52 output[WORD_RESULT_ID] = VNA_NO_CONNECTION_ID;
53 output[WORD_RESULT_MSG] = VNA_NO_CONNECTION_MSG;
54 break;
55 }
56 }
57 }
58
59 output.push_back({WORD_RESULT_DATA, device_results});
60 return output;
61}

294json TaskManager::proceed_task(const json &task) {
295 json result;
296
297 if (task[WORD_TASK_TYPE] == TASK_TYPE_CONNECT) {
298 result[WORD_RESULT] = connect_task(task[WORD_TASK_ARGS]);
299 } else if (task[WORD_TASK_TYPE] == TASK_TYPE_DISCONNECT) {
300 disconnect_task();
301
302 result[WORD_RESULT] = {
303 {WORD_RESULT_ID, RESULT_OK_ID},
304 {WORD_RESULT_MSG, RESULT_OK_MSG},
305 {WORD_RESULT_DATA, true}
306 };
307 } else if (task[WORD_TASK_TYPE] == TASK_TYPE_CONFIGURE) {
308 bool task_result = configure_task(task[WORD_TASK_ARGS]);
309
310 result[WORD_RESULT] = {
311 {WORD_RESULT_ID, task_result ? RESULT_OK_ID : VNA_CONFIGURE_ERR_ID},
312 {WORD_RESULT_MSG, task_result ? RESULT_OK_MSG : VNA_CONFIGURE_ERR_MSG},
313 {WORD_RESULT_DATA, task_result}
314 };
315 } else if (task[WORD_TASK_TYPE] == TASK_TYPE_SET_POWER) {
316 bool task_result = set_power_task(task[WORD_TASK_ARGS]);
317
318 result[WORD_RESULT] = {
319 {WORD_RESULT_ID, task_result ? RESULT_OK_ID : ERR_SET_POWER_ID},
320 {WORD_RESULT_MSG, task_result ? RESULT_OK_MSG : ERR_SET_POWER_MSG},
321 {WORD_RESULT_DATA, task_result}
322 };
323 } else if (task[WORD_TASK_TYPE] == TASK_TYPE_SET_FREQ) {
324 bool task_result = set_freq_task(task[WORD_TASK_ARGS]);
325
326 result[WORD_RESULT] = {
327 {WORD_RESULT_ID, task_result ? RESULT_OK_ID : ERR_SET_FREQ_ID},
328 {WORD_RESULT_MSG, task_result ? RESULT_OK_MSG : ERR_SET_FREQ_MSG},
329 {WORD_RESULT_DATA, task_result}
330 };
331 } else if (task[WORD_TASK_TYPE] == TASK_TYPE_SET_FREQ_RANGE) {
332 bool task_result = set_freq_range_task(task[WORD_TASK_ARGS]);
333
334 result[WORD_RESULT] = {
335 {WORD_RESULT_ID, task_result ? RESULT_OK_ID : ERR_SET_FREQ_RANGE_ID},
336 {WORD_RESULT_MSG, task_result ? RESULT_OK_MSG : ERR_SET_FREQ_RANGE_MSG},
337 {WORD_RESULT_DATA, task_result}
338 };
339 } else if (task[WORD_TASK_TYPE] == TASK_TYPE_SET_ANGLE) {
340 bool task_result = set_angle_task(task[WORD_TASK_ARGS]);
341
342 result[WORD_RESULT] = {
343 {WORD_RESULT_ID, task_result ? RESULT_OK_ID : ERR_SET_ANGLE_ID},
344 {WORD_RESULT_MSG, task_result ? RESULT_OK_MSG : ERR_SET_ANGLE_MSG},
345 {WORD_RESULT_DATA, task_result}
346 };
347 } else if (task[WORD_TASK_TYPE] == TASK_TYPE_SET_ANGLE_RANGE) {
348 bool task_result = set_angle_range_task(task[WORD_TASK_ARGS]);
349
350 result[WORD_RESULT] = {
351 {WORD_RESULT_ID, task_result ? RESULT_OK_ID : ERR_SET_ANGLE_RANGE_ID},
352 {WORD_RESULT_MSG, task_result ? RESULT_OK_MSG : ERR_SET_ANGLE_RANGE_MSG},
353 {WORD_RESULT_DATA, task_result}
354 };
355 } else if (task[WORD_TASK_TYPE] == TASK_TYPE_CHANGE_PATH) {
356 bool task_result = set_path_task(task[WORD_TASK_ARGS]);
357
358 result[WORD_RESULT] = {
359 {WORD_RESULT_ID, task_result ? RESULT_OK_ID : ERR_CHANGE_SWITCH_PATH_ID},

361 {WORD_RESULT_DATA, task_result}
362 };
363 } else if (task[WORD_TASK_TYPE] == TASK_TYPE_GET_DATA) {
364 std::string task_result = get_data_task(task[WORD_TASK_ARGS]);

367 {WORD_RESULT_ID, !task_result.empty() ? RESULT_OK_ID : ERR_GETTING_DATA_ID},

377 }
378

381 result[WORD_RESULT][WORD_RESULT_ID] == 0 ? "Task completed" : "Can't proceed task. Error code: {}",
382 to_string(result[WORD_RESULT][WORD_RESULT_ID]));

385}

402json TaskManager::proceed_task_list(const json &task_list) {
403 json result;
404 json nested_result;
405
406 std::vector<json> nested_task_list{};
407 std::vector<json> task_cache = task_list.get<std::vector<json>>();
408
409 for (const json &task : task_cache) {
410 logger::log(LEVEL_TRACE, "Preparing task = {}", to_string(task));
411
412 if (stop_requested) {
413 logger::log(LEVEL_WARN, "Task list proceeding stopped");
414 device_set.reset_stop_request();
415 stop_requested = false;
416
417 result[WORD_RESULT] = {
418 {WORD_RESULT_ID, MEASUREMENTS_STOPS_ID},
419 {WORD_RESULT_MSG, MEASUREMENTS_STOPS_MSG},
420 {WORD_RESULT_DATA, json::value_t::null}
421 };
422
423 return result;
424 }
425
426 if (!task.contains(WORD_NESTED)) {
427 result = proceed_task(task);
428
429 if (result[WORD_RESULT][WORD_RESULT_ID] != 0) {
430 return result;
431 }
432 } else if (task.contains(WORD_NESTED) && (task[WORD_TASK_TYPE] == TASK_TYPE_CONNECT ||
433 task[WORD_TASK_TYPE] == TASK_TYPE_CONFIGURE || task[WORD_TASK_TYPE] == TASK_TYPE_SET_POWER ||
434 task[WORD_TASK_TYPE] == TASK_TYPE_SET_ANGLE || task[WORD_TASK_TYPE] == TASK_TYPE_SET_FREQ ||
435 task[WORD_TASK_TYPE] == TASK_TYPE_DISCONNECT)) {
436 logger::log(
437 LEVEL_WARN,
438 R"(Task '{}' has arg 'nested', but this task cannot be nested. This arg ignored.)",
439 task[WORD_TASK_TYPE].get<std::string>());
440
441 logger::log(LEVEL_TRACE, "Processing task without 'nested' argument");
442 result = proceed_task(task);
443
444 if (result[WORD_RESULT][WORD_RESULT_ID] != 0) {
445 return result;
446 }
447 } else {
448 nested_task_list.push_back(task);
449 }
450 }
451
452 if (!result.is_null() && result[WORD_RESULT][WORD_RESULT_ID] != 0) {
453 return result;
454 }
455
456 std::sort(nested_task_list.begin(), nested_task_list.end(), array_utils::compare_nested);
457 logger::log(LEVEL_TRACE, "Nested task list size = {}", nested_task_list.size());
458
459 nested_result = proceed_nested_task_list(std::move(nested_task_list));
460
461 if (nested_result[WORD_RESULT][WORD_RESULT_ID] != 0 || result.is_null()) {
462 return nested_result;
463 } else {
464 result[WORD_RESULT][WORD_RESULT_DATA] = nested_result[WORD_RESULT][WORD_RESULT_DATA];
465 }
466
467 return result;
468}

477json TaskManager::proceed_nested_task_list(std::vector<json> nested_task_list) {
478 json result;
479 std::string data{};
480 std::string acquired_data{};
481
482 for (json &nested_task : nested_task_list) {
483 logger::log(LEVEL_TRACE, "Preparing nested task = {}", to_string(nested_task));
484
485 if (stop_requested) {
486 logger::log(LEVEL_WARN, "Nested task list proceeding stopped");
487 device_set.reset_stop_request();
488 stop_requested = false;
489
490 result[WORD_RESULT] = {
491 {WORD_RESULT_ID, MEASUREMENTS_STOPS_ID},
492 {WORD_RESULT_MSG, MEASUREMENTS_STOPS_MSG},
493 {WORD_RESULT_DATA, json::value_t::null}
494 };
495
496 return result;
497 }
498
499 if (nested_task[WORD_TASK_TYPE] == TASK_TYPE_GET_DATA) {
500 continue;
501 } else if (nested_task[WORD_TASK_TYPE] == TASK_TYPE_SET_ANGLE_RANGE) {
502 logger::log(LEVEL_DEBUG, to_string(nested_task[WORD_TASK_ARGS]));
503 result = proceed_task(nested_task);
504
505 if (result[WORD_RESULT][WORD_RESULT_ID] != 0) {
506 return result;
507 }
508
509 nested_task[WORD_TASK_TYPE] = TASK_TYPE_NEXT_ANGLE;
510 nested_task[WORD_AXIS] = nested_task[WORD_TASK_ARGS][WORD_AXIS];
511
512 nested_task.erase(WORD_TASK_ARGS);
513 } else if (nested_task[WORD_TASK_TYPE] == TASK_TYPE_SET_FREQ_RANGE) {
514 result = proceed_task(nested_task);
515
516 if (result[WORD_RESULT][WORD_RESULT_ID] != 0) {
517 return result;
518 }
519
520 nested_task[WORD_TASK_TYPE] = TASK_TYPE_NEXT_FREQ;
521 nested_task.erase(WORD_TASK_ARGS);
522 }
523 }
524
525 for (int nested_pos = 0; nested_pos < nested_task_list.size(); ++nested_pos) {
526 if (stop_requested) {
527 logger::log(LEVEL_WARN, "Nested task list proceeding stopped");
528 device_set.reset_stop_request();
529 stop_requested = false;
530
531 result[WORD_RESULT] = {
532 {WORD_RESULT_ID, MEASUREMENTS_STOPS_ID},
533 {WORD_RESULT_MSG, MEASUREMENTS_STOPS_MSG},
534 {WORD_RESULT_DATA, json::value_t::null}
535 };
536
537 return result;
538 }
539
540 if (nested_task_list[nested_pos][WORD_TASK_TYPE] == TASK_TYPE_GET_DATA) {
541 acquired_data = std::move(get_data_task(nested_task_list[nested_pos][WORD_TASK_ARGS]));
542
543 if (stop_requested) {
544 logger::log(LEVEL_WARN, "Nested task list proceeding stopped");
545 device_set.reset_stop_request();
546 stop_requested = false;
547
548 result[WORD_RESULT] = {
549 {WORD_RESULT_ID, MEASUREMENTS_STOPS_ID},
550 {WORD_RESULT_MSG, MEASUREMENTS_STOPS_MSG},
551 {WORD_RESULT_DATA, json::value_t::null}
552 };
553
554 return result;
555 }
556
557 if (!acquired_data.empty()) {
558 data += (data.empty() ? "" : ";") + acquired_data;
559 } else {
560 result[WORD_RESULT] = {
561 {WORD_RESULT_ID, ERR_GETTING_DATA_ID},
562 {WORD_RESULT_MSG, ERR_GETTING_DATA_MSG},
563 {WORD_RESULT_DATA, data}
564 };
565
566 return result;
567 }
568 } else if (nested_task_list[nested_pos][WORD_TASK_TYPE] == TASK_TYPE_NEXT_FREQ) {
569 if (!device_set.is_using_ext_gen()) {
570 continue;
571 } else {
572 switch (next_freq_task()) {
573 case FREQ_MOVE_OK:
574 nested_pos = -1;
575 continue;
576 case FREQ_MOVE_BOUND:
577 if (device_set.move_to_start_freq()) {
578 continue;
579 } else {
580 result[WORD_RESULT] = {
581 {WORD_RESULT_ID, ERR_SET_FREQ_ID},
582 {WORD_RESULT_MSG, ERR_SET_FREQ_MSG},
583 {WORD_RESULT_DATA, false}
584 };
585 }
586 default:
587 result[WORD_RESULT] = {
588 {WORD_RESULT_ID, ERR_SET_FREQ_ID},
589 {WORD_RESULT_MSG, ERR_SET_FREQ_MSG},
590 {WORD_RESULT_DATA, false}
591 };
592
593 return result;
594 }
595 }
596 } else if (nested_task_list[nested_pos][WORD_TASK_TYPE] == TASK_TYPE_NEXT_ANGLE) {
597 switch (next_angle_task(nested_task_list[nested_pos][WORD_AXIS].get<int>())) {
598 case ANGLE_MOVE_OK:
599 nested_pos = -1;
600 continue;
601 case ANGLE_MOVE_BOUND:
602 if (device_set.move_to_start_angle(nested_task_list[nested_pos][WORD_AXIS].get<int>())) {
603 continue;
604 } else {
605 result[WORD_RESULT] = {
606 {WORD_RESULT_ID, ERR_SET_ANGLE_ID},
607 {WORD_RESULT_MSG, ERR_SET_ANGLE_MSG},
608 {WORD_RESULT_DATA, false}
609 };
610 }
611 default:
612 result[WORD_RESULT] = {
613 {WORD_RESULT_ID, ERR_SET_ANGLE_ID},
614 {WORD_RESULT_MSG, ERR_SET_ANGLE_MSG},
615 {WORD_RESULT_DATA, false}
616 };
617
618 return result;
619 }
620 }
621 }
622
623 result[WORD_RESULT] = {
624 {WORD_RESULT_ID, RESULT_OK_ID},
625 {WORD_RESULT_MSG, RESULT_OK_MSG},
626 {WORD_RESULT_DATA, data}
627 };
628
629 return result;
630}

643std::string TaskManager::proceed(const json &data) {
644 json answer;
645
646 auto start_time = std::chrono::high_resolution_clock::now();
647
648 if (data.contains(WORD_TASK)) {
649 logger::log(LEVEL_INFO, "Received task");
650 answer = proceed_task(data[WORD_TASK]);
651 } else if (data.contains(WORD_TASK_LIST)) {
652 logger::log(LEVEL_INFO, "Received task list");
653 answer = proceed_task_list(data[WORD_TASK_LIST]);
654 } else {
655 answer = {
656 WORD_RESULT, {
657 {WORD_RESULT_ID, NO_TASK_ID},
658 {WORD_RESULT_MSG, NO_TASK_MSG}
659 }
660 };
661 }
662
663 auto stop_time = std::chrono::high_resolution_clock::now();
664 auto duration = std::chrono::duration_cast<std::chrono::milliseconds>(stop_time - start_time).count();
665
666 auto duration_ms = duration % 1000;
667 duration = (long long) ((duration - duration_ms) / 1000);
668
669 auto duration_s = duration % 60;
670 duration = (long long) ((duration - duration_s) / 60);
671
672 logger::log(LEVEL_INFO, "Proceeding finished for {}:{:02}.{:03}", duration, duration_s, duration_ms);
673
674 return to_string(answer);
675}
