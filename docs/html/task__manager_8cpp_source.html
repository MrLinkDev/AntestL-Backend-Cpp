489std::string TaskManager::proceed(const json &data) {
490 json answer;
491
492 auto start_time = std::chrono::high_resolution_clock::now();
493
494 if (data.contains(WORD_TASK)) {
495 logger::log(LEVEL_INFO, "Received task");
496 answer = proceed_task(data[WORD_TASK]);
497 } else if (data.contains(WORD_TASK_LIST)) {
498 logger::log(LEVEL_INFO, "Received task list");
499 answer = proceed_task_list(data[WORD_TASK_LIST]);
500 } else {
501 answer = {
502 WORD_RESULT, {
503 {WORD_RESULT_ID, NO_TASK_ID},
504 {WORD_RESULT_MSG, NO_TASK_MSG}
505 }
506 };
507 }
508
509 auto stop_time = std::chrono::high_resolution_clock::now();
510 auto duration = std::chrono::duration_cast<std::chrono::milliseconds>(stop_time - start_time).count();
511
512 auto duration_ms = duration % 1000;
513 duration = (long long) ((duration - duration_ms) / 1000);
514
515 auto duration_s = duration % 60;
516 duration = (long long) ((duration - duration_s) / 60);
517
518 logger::log(LEVEL_INFO, "Proceeding finished for {}:{:02}.{:03}", duration, duration_s, duration_ms);
519
520 return to_string(answer);
521}
