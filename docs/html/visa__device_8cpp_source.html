46std::string VisaDevice::read() {
47 std::string data{};
48 ViChar buffer[BUFFER_SIZE];
49
50 while (true) {
51 status = viRead(device, reinterpret_cast<ViPBuf>(buffer), BUFFER_SIZE, ret_count);
52
53 if (status < VI_SUCCESS) {
54 return std::string{};
55 }
56
57 for (char b : buffer) {
58 if (b == device_config.termination) {
59 logger::log(LEVEL_TRACE, "READ: {}", data);
60 return data;
61 }
62
63 data.push_back(b);
64 }
65
66 memset(buffer, 0, sizeof(buffer));
67 }
68}

135void VisaDevice::connect() {
136 status = viOpenDefaultRM(&resource_manager);
137
138 if (status < VI_SUCCESS) {
139 logger::log(LEVEL_ERROR, "Can't open resource manager");
140
141 connected = false;
142 return;
143 }
144
145 logger::log(LEVEL_TRACE, "Connecting to device with address {}", device_config.address);
146
147 status = viOpen(
148 resource_manager,
149 device_config.address.c_str(),
150 VI_NULL,
151 VI_NULL,
152 &device);
153
154 if (status < VI_SUCCESS) {
155 connected = false;
156 return;
157 } else {
158 logger::log(LEVEL_DEBUG, "Connected to device with address {}", device_config.address);
159 }
160
161 status = viSetAttribute(device, VI_ATTR_TMO_VALUE, DEFAULT_TIMEOUT);
162
163 if (status < VI_SUCCESS) {
164 logger::log(LEVEL_ERROR, "Can't set timeout");
165
166 connected = false;
167 return;
168 }
169
170 status = viSetAttribute(device, VI_ATTR_TERMCHAR, DEFAULT_VISA_TERM);
171
172 if (status < VI_SUCCESS) {
173 logger::log(LEVEL_ERROR, "Can't set termination character");
174
175 connected = false;
176 return;
177 }
178
179 status = viSetAttribute(device, VI_ATTR_TERMCHAR_EN, true);
180
181 if (status < VI_SUCCESS) {
182 logger::log(LEVEL_ERROR, "Can't enable termination character");
183
184 connected = false;
185 return;
186 }
187
188 connected = true;
189}

236

238

322std::string VisaDevice::send(std::string command, bool read_data) {
323 std::string data{};
324
325 if (command[command.length() - 1] == '?') {
326 data = query(command);
327
328 if (data.empty()) {
329 logger::log(LEVEL_ERROR, READ_ERROR_MSG);
330 throw antestl_exception(READ_ERROR_MSG, READ_ERROR_CODE);
331 }
332 } else {
333 if (write(command) == FAILURE) {
334 logger::log(LEVEL_ERROR, WRITE_ERROR_MSG);
335 throw antestl_exception(WRITE_ERROR_MSG, WRITE_ERROR_CODE);
336 }
337
338 if (read_data) {
339 data = read();
340

342 logger::log(LEVEL_ERROR, READ_ERROR_MSG);
343 throw antestl_exception(READ_ERROR_MSG, READ_ERROR_CODE);
344 }
345 }
346 }
347
348 return data;
