131void KeysightM9807A::create_traces(std::vector<int> port_list, bool external) {
132 send_wait_err(":CALCulate:PARameter:DELete:ALL");
133
134 std::string trace_name{};
135 std::string trace_params{};
136
137 int trace_arg = external ? 0 : source_port;
138
139 for (int pos = 0; pos < port_list.size(); ++pos) {
140 if (external) {
141 char port_name = port_names[
142 array_utils::index(
143 port_numbers,
144 M9807A_PORT_COUNT,
145 port_list[pos]
146 )];
147
148 char source_port_name = port_names[
149 array_utils::index(
150 port_numbers,
151 M9807A_PORT_COUNT,
152 port_list[source_port]
153 )];
154
155 if (meas_type == MEAS_TRANSITION) {
156 trace_name = std::format("{}/{}", port_name, source_port_name);
157 trace_params = std::format("{},{}", trace_name, trace_arg);
158 } else {
159 trace_name = trace_params = std::format("S{}{}", port_list[pos], port_list[pos]);
160 }
161 } else {
162 if (meas_type == MEAS_TRANSITION) {
163 trace_name = trace_params = std::format("S{}{}", port_list[pos], source_port);
164 } else {
165 trace_name = trace_params = std::format("S{}{}", port_list[pos], port_list[pos]);
166 }
167 }
168
169 send_wait_err(R"(:CALCulate1:CUSTom:DEFine "TR{}","Standard","{}")", port_list[pos], trace_params);
170 send_wait_err(":DISPlay:WINDow:TRACe{}:FEED \"TR{}\"", port_list[pos], port_list[pos]);
171 }
172}
