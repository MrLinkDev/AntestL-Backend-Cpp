221 int wait_client() {

272 int read_data(std::string &data) {
273 char buffer[DEFAULT_BUFFER_SIZE] = { 0 };
274 bool finish = false;
275
276 while (true) {
277 if (recv(client, buffer, sizeof(buffer), 0) == SOCKET_ERROR) {
278 return DATA_READ_ERROR;
279 }
280
281 for (int pos = 0; pos < DEFAULT_BUFFER_SIZE; ++pos) {
282 if (buffer[pos] == termination[0]) {
283 finish = true;
284 } else if (buffer[pos] == termination[1] && finish) {
285 memset(buffer, 0, sizeof(buffer));
286
287 logger::log(LEVEL_INFO, "Data was read from client");
288 logger::log(LEVEL_DEBUG, data);
289
290 return DATA_READ_OK;
291 } else if (buffer[pos] == '\0') {
292 continue;
293 } else {
294 if (finish) {
295 finish = false;
296 }
297
298 data += buffer[pos];
299 }
300 }
301
302 memset(buffer, 0, sizeof(buffer));
303 }
304 }
