48 std::vector<std::string> address_list = string_utils::split(device_addresses, ADDRESS_DELIMITER);
49
50 for (const std::string &address : address_list) {

53 axis.send("PROMPT 0");
54 axis.send("CLRFAULT");

61 axis.send("MH");
