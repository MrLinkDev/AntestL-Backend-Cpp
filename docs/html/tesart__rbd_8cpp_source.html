86TesartRbd::TesartRbd(const std::string &device_addresses) {
87 std::vector<std::string> address_list = string_utils::split(device_addresses, ADDRESS_DELIMITER);
88
89 for (int i = 0; i < address_list.size(); ++i) {
90 logger::log(LEVEL_TRACE, "Address [{}] = {}", i, address_list[i]);
91 }
92
93 for (const std::string &address : address_list) {
94 axes.push_back(VisaDevice(address));
95
96 axes[axes.size() - 1].connect();
97
98 axes[axes.size() - 1].send("PROMPT 0\r");
99 axes[axes.size() - 1].send("CLRFAULT\r");
100 axes[axes.size() - 1].send("EN\r");
101
102 axes[axes.size() - 1].clear();
103 std::this_thread::sleep_for(TESART_RBD_SLEEP_TIME_INIT);
104
105 if (!(status(&axes[axes.size() - 1]) & BIT_REF_SET)) {
106 axes[axes.size() - 1].send("MH\r");
107 }
108 }
109
110 init_params(axes.size());
111}

155bool TesartRbd::is_stopped(int axis_num) {
156 int status_code = status(axis_num);
157
158 Logger::log(LEVEL_TRACE, "IN POS = {} ({})", bool(status_code & BIT_IN_POS), status_code & BIT_IN_POS);
159 Logger::log(LEVEL_TRACE, "MOVE BLOCK = {} ({})", bool(status_code & BIT_MOVE_BLOCK), status_code & BIT_MOVE_BLOCK);
160
161 return bool(status_code & BIT_IN_POS) and !bool(status_code & BIT_MOVE_BLOCK);
162}

249void TesartRbd::set_angle_range(float start_angle, float stop_angle, int points, int axis_num) {
250 this->start_angle[axis_num] = start_angle;
251 this->stop_angle[axis_num] = stop_angle;
252
253 this->points[axis_num] = points;
254
255 angle_step[axis_num] =
256 this->points[axis_num] <= 1 ?
257 0 : (this->stop_angle[axis_num] - this->start_angle[axis_num]) / (this->points[axis_num] - 1);
258
259 this->current_angle[axis_num] = start_angle;
260 this->current_point[axis_num] = 0;
