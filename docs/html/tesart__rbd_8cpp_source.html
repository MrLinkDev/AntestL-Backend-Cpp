80TesartRbd::TesartRbd(const std::string &device_addresses) {
81 std::vector<std::string> address_list = string_utils::split(device_addresses, ADDRESS_DELIMITER);
82
83 for (const std::string &address : address_list) {
84 VisaDevice axis(address);
85
86 axis.send("PROMPT 0");
87 axis.send("CLRFAULT");
88 axis.send("EN");
89
90 axis.clear();
91 std::this_thread::sleep_for(TESART_RBD_SLEEP_TIME_INIT);
92
93 if (!(status(&axis) & BIT_REF_SET)) {
94 axis.send("MH");
95 }
96
97 axes.push_back(axis);
98 }
99
100 init_params();
101}

231void TesartRbd::set_angle_range(float start_angle, float stop_angle, int points, int axis_num) {
232 this->start_angle[axis_num] = start_angle;
233 this->stop_angle[axis_num] = stop_angle;
234
235 this->points[axis_num] = points;
236
237 angle_step[axis_num] =
238 this->points[axis_num] <= 1 ?
239 0 : (this->stop_angle[axis_num] - this->start_angle[axis_num]) / (this->points[axis_num] - 1);
240
241 this->current_angle[axis_num] = start_angle;
242 this->current_point[axis_num] = 0;

264int TesartRbd::next_angle(int axis_num) {
