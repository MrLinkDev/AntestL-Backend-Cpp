49static std::string get_current_time() {
50 struct timeval time_value{};
51 struct tm *time_struct;
52
53 gettimeofday(&time_value, NULL);
54 int milliseconds = lrint(time_value.tv_usec / 1000.0);
55
56 if (milliseconds >= 1000) {
57 milliseconds -= 1000;
58 time_value.tv_sec++;
59 }
60
61 time_t current_time = time_value.tv_sec;
62 time_struct = localtime(&current_time);
63
64 char str_time[DEFAULT_BUFFER_SIZE];
65 strftime(str_time, DEFAULT_BUFFER_SIZE, DEFAULT_TIME_FORMAT, time_struct);
66
67 return std::string(str_time) + std::format(".{:03}", milliseconds);
68}
