52static std::string get_current_time() {
53 struct timeval time_value{};
54 struct tm *time_struct;
55
56 gettimeofday(&time_value, NULL);
57 int milliseconds = lrint(time_value.tv_usec / 1000.0);
58
59 if (milliseconds >= 1000) {
60 milliseconds -= 1000;
61 time_value.tv_sec++;
62 }
63
64 time_t current_time = time_value.tv_sec;
65 time_struct = localtime(&current_time);
66
67 char str_time[DEFAULT_BUFFER_SIZE];
68 strftime(str_time, DEFAULT_BUFFER_SIZE, DEFAULT_TIME_FORMAT, time_struct);
69
70 return std::string(str_time) + std::format(".{:03}", milliseconds);
71}
