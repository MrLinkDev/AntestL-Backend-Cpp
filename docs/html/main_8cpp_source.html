151void task_server_thread_f(std::stop_token s_token) {
152 if (task_server.create() != SOCKET_CREATED && !stop_process) {
153 exit(1);
154 }
155
156 if (task_server.wait_client() != CLIENT_CONNECTED && !stop_process) {
157 exit(1);
158 }
159
160 while (!s_token.stop_requested() && !stop_process && !wait_another) {
161 try {
162 input_data = std::move(json::parse(task_server.read_data()));
163 } catch (const json::parse_error &err) {
164 if (!task_server.is_connected()) {
165 task_thread->request_stop();
166 data_thread->request_stop();
167
168 wait_another = true;
169
181 } else {
182 logger::log(LEVEL_ERROR, "Seems like input data cannot be parsed into json. Check input data!");
183
184 std::this_thread::sleep_for(50ms);
185 continue;
186 }
187 }
188
189 if (task_manager.received_stop_task(input_data)) {
190 task_manager.request_stop();
191 continue;
192 }
193
194 if (task_manager.received_disconnect_task(input_data)) {
195 task_thread->request_stop();
196 data_thread->request_stop();
197 }
198
199 std::unique_lock u_lk(mtx);
200 cv.wait(u_lk, []{return processed;});
201
202 data_buffer = std::move(input_data);
203
204 received = true;
205 processed = false;
206
207 u_lk.unlock();
208 cv.notify_one();
209 }
210
211 task_server.close();
212}

219void data_server_thread_f(std::stop_token s_token) {
220 if (data_server.create() != SOCKET_CREATED && !stop_process) {
221 exit(1);
222 }
223
224 if (data_server.wait_client() != CLIENT_CONNECTED && !stop_process) {
225 exit(1);
226 }
227
228 std::string result{};
229
230 while (!s_token.stop_requested() && !stop_process && !wait_another) {
231 std::unique_lock u_lk(mtx);
232 cv.wait(u_lk, []{return received;});
233
234 if (data_server.is_connected() && !wait_another) {
235 result = std::move(task_manager.proceed(data_buffer));
236 data_server.send_data(result);
237 }
238
239 processed = true;
240 received = false;
241
242 u_lk.unlock();
243 cv.notify_one();
244 }
245
246 data_server.close();
247}
