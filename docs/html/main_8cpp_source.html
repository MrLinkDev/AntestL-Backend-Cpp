141void task_server_thread_f(std::stop_token s_token) {
142 if (task_server.create() != SOCKET_CREATED && !stop_process) {
143 exit(1);
144 }
145
146 if (task_server.wait_client() != CLIENT_CONNECTED && !stop_process) {
147 exit(1);
148 }
149
150 while (!s_token.stop_requested() || !stop_process) {
151 try {
152 input_data = std::move(json::parse(task_server.read_data()));
153 } catch (const json::parse_error &err) {
154 logger::log(LEVEL_ERROR, "Seems like input data cannot be parsed into json. Check input data!");
155
156 std::this_thread::sleep_for(50ms);
157 continue;
158 }
159
160 if (task_manager.received_stop_task(input_data)) {
161 task_manager.request_stop();
162 continue;
163 }
164
165 if (task_manager.received_disconnect_task(input_data)) {
166 task_thread->request_stop();
167 data_thread->request_stop();
168 }
169
170 std::unique_lock u_lk(mtx);
171 cv.wait(u_lk, []{return processed;});
172
173 data_buffer = std::move(input_data);
174
175 received = true;
176 processed = false;
177
178 u_lk.unlock();
179 cv.notify_one();
180 }
181}

188void data_server_thread_f(std::stop_token s_token) {
189 if (data_server.create() != SOCKET_CREATED && !stop_process) {
190 exit(1);
191 }
192
193 if (data_server.wait_client() != CLIENT_CONNECTED && !stop_process) {
194 exit(1);
195 }
196
197 std::string result{};
198
199 while (!s_token.stop_requested() || !stop_process) {
200 std::unique_lock u_lk(mtx);
201 cv.wait(u_lk, []{return received;});
202
203 result = std::move(task_manager.proceed(data_buffer));
204 data_server.send_data(result);
205
206 processed = true;
207 received = false;
208
209 u_lk.unlock();
210 cv.notify_one();
211 }
212
213 data_server.close();
214}
