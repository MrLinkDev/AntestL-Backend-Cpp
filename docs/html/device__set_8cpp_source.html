669data_t DeviceSet::get_data(std::vector<int> port_list) {
670 if (stop_requested) {
671 logger::log(LEVEL_WARN, "Device set stops measuring");
672 stop_requested = false;
673
674 return data_t{};
675 }
676
677 logger::log(LEVEL_TRACE, "Preparing to acquire data");
678 data_t acquired_data{};
679
680 try {
681 if (!traces_configured) {
682 vna->create_traces(port_list, using_ext_gen);
683 traces_configured = true;
684 }
685
686 logger::log(LEVEL_TRACE, "Traces created");
687 } catch (int error_code) {
688 logger::log(LEVEL_ERROR, "Can't create traces");
689 return acquired_data;
690 }
691
692 for (int port_pos = 0; port_pos < port_list.size(); ++port_pos) {
693 if (stop_requested) {
694 logger::log(LEVEL_WARN, "Device set stops measuring");
695 stop_requested = false;
696
697 return data_t{};
698 }
699
700 int port_num = port_list[port_pos];
701 logger::log(LEVEL_TRACE, "Port = {}", port_num);
702
703 if (meas_type == MEAS_TRANSITION && port_pos == 0) {
704 try {
705 if (using_ext_gen) {
706 ext_gen->rf_on();
707 } else {
708 vna->rf_on(vna->get_source_port());
709 }
710
711 logger::log(LEVEL_TRACE, "Source port enabled");
712 } catch (int error_code) {
713 logger::log(LEVEL_ERROR, "Can't enable source port");
714 return acquired_data;
715 }
716 } else if (meas_type == MEAS_REFLECTION) {
717 try {
718 if (using_ext_gen) {
719 ext_gen->rf_on();
720 } else {
721 vna->rf_on(port_num);
722 }
723
724 logger::log(LEVEL_TRACE, "Port {} enabled", port_num);
725 } catch (int error_code) {
726 logger::log(LEVEL_ERROR, "Can't enable port {}", port_num);
727 return acquired_data;
728 }
729 }
730
731 if ((port_pos == 0 && meas_type == MEAS_TRANSITION) || meas_type == MEAS_REFLECTION) {
732 try {
733 vna->trigger();
734 vna->init();
735
736 logger::log(LEVEL_TRACE, "Measurements restarted");
737 } catch (int error_code) {
738 logger::log(LEVEL_ERROR, "Can't restart measurement");
739 return acquired_data;
740 }
741 }
742
743 try {
744 acquired_data.insert_iq_port_data(vna->get_data(port_pos));
745 logger::log(LEVEL_TRACE, "Data for port {} acquired", port_num);
746 } catch (int error_code) {
747 logger::log(LEVEL_ERROR, "Can't acquire data for port {} from VNA", port_num);
748 return acquired_data;
749 }
750
751 if (meas_type == MEAS_TRANSITION && port_pos == port_list.size() - 1) {
752 try {
753 if (using_ext_gen) {
754 ext_gen->rf_off();
755 } else {
756 vna->rf_off(vna->get_source_port());
757 }
758
759 logger::log(LEVEL_TRACE, "Source port disabled");
760 } catch (int error_code) {
761 logger::log(LEVEL_ERROR, "Can't disable source port");
762 return acquired_data;
763 }
764 } else if (meas_type == MEAS_REFLECTION) {
765 try {
766 if (using_ext_gen) {
767 ext_gen->rf_off();
768 } else {
769 vna->rf_off(port_num);
770 }
771
772 logger::log(LEVEL_TRACE, "Port {} disabled", port_num);
773 } catch (int error_code) {
774 logger::log(LEVEL_ERROR, "Can't disable port {}", port_num);
775 return acquired_data;
776 }
777 }
778 }
779
780 acquired_data.insert_angles(get_current_angles());
781
782 if (using_ext_gen) {
783 acquired_data.insert_freq(get_current_freq());
784 } else {
785 acquired_data.insert_freq_list(get_freq_list());
786 }
787
788 logger::log(LEVEL_DEBUG, "Data acquired");
789
790 return acquired_data;
791}
