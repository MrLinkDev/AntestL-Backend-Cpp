31bool DeviceSet::connect(int device_type, std::string device_model, const std::string &device_address) {
32 std::transform(device_model.begin(), device_model.end(), device_model.begin(), ::toupper);
33
34 try {
35 switch (device_type) {
36 case DEVICE_VNA:
37 logger::log(LEVEL_TRACE, "Connecting to VNA");

39 if (device_model == "M9807A") {
40 vna = new KeysightM9807A(device_address);

42 vna = new PlanarS50244(device_address);

44 return false;

48 return vna->is_connected();

52

55 logger::log(LEVEL_TRACE, "Connecting to RBD");
56
57 if (device_model == "TESART_RBD") {
58 rbd = new TesartRbd(device_address);
59 } else if (device_model == "UPKB_RBD") {
60 //TODO: Добавить инициализацию объекта

62 rbd = new DemoRbd(device_address);
63 } else {
64 return false;
65 }
66
67 return rbd->is_connected();
68 default:
69 return false;
70 }
71 } catch (const antestl_exception &exception) {

676data_t DeviceSet::get_data(std::vector<int> port_list) {
677 if (stop_requested) {
678 logger::log(LEVEL_WARN, "Device set stops measuring");
679 stop_requested = false;
680
681 return data_t{};
682 }
683
684 logger::log(LEVEL_TRACE, "Preparing to acquire data");
685 data_t acquired_data{};
686
687 try {
688 if (!traces_configured) {
689 vna->create_traces(port_list, using_ext_gen);
690 traces_configured = true;
691 }
692
693 logger::log(LEVEL_TRACE, "Traces created");
694 } catch (int error_code) {
695 logger::log(LEVEL_ERROR, "Can't create traces");
696 return acquired_data;
697 }
698
699 for (int port_pos = 0; port_pos < port_list.size(); ++port_pos) {
700 if (stop_requested) {
701 logger::log(LEVEL_WARN, "Device set stops measuring");
702 stop_requested = false;
703
704 return data_t{};
705 }
706
707 int port_num = port_list[port_pos];
708 logger::log(LEVEL_TRACE, "Port = {}", port_num);
709
710 if (meas_type == MEAS_TRANSITION && port_pos == 0) {
711 try {
712 if (using_ext_gen) {
713 ext_gen->rf_on();
714 } else {
715 vna->rf_on(vna->get_source_port());
716 }
717
718 logger::log(LEVEL_TRACE, "Source port enabled");
719 } catch (int error_code) {
720 logger::log(LEVEL_ERROR, "Can't enable source port");
721 return acquired_data;
722 }
723 } else if (meas_type == MEAS_REFLECTION) {
724 try {
725 if (using_ext_gen) {
726 ext_gen->rf_on();
727 } else {
728 vna->rf_on(port_num);
729 }
730
731 logger::log(LEVEL_TRACE, "Port {} enabled", port_num);
732 } catch (int error_code) {
733 logger::log(LEVEL_ERROR, "Can't enable port {}", port_num);
734 return acquired_data;
735 }
736 }
737
738 if ((port_pos == 0 && meas_type == MEAS_TRANSITION) || meas_type == MEAS_REFLECTION) {
739 try {
740 vna->trigger();
741 vna->init();
742
743 logger::log(LEVEL_TRACE, "Measurements restarted");
744 } catch (int error_code) {
745 logger::log(LEVEL_ERROR, "Can't restart measurement");
746 return acquired_data;
747 }
748 }
749
750 try {
751 acquired_data.insert_iq_port_data(vna->get_data(port_pos));
752 logger::log(LEVEL_TRACE, "Data for port {} acquired", port_num);
753 } catch (int error_code) {
754 logger::log(LEVEL_ERROR, "Can't acquire data for port {} from VNA", port_num);
755 return acquired_data;
756 }
757
758 if (meas_type == MEAS_TRANSITION && port_pos == port_list.size() - 1) {
759 try {
760 if (using_ext_gen) {
761 ext_gen->rf_off();
762 } else {
763 vna->rf_off(vna->get_source_port());
764 }
765
766 logger::log(LEVEL_TRACE, "Source port disabled");
767 } catch (int error_code) {
768 logger::log(LEVEL_ERROR, "Can't disable source port");
769 return acquired_data;
770 }
771 } else if (meas_type == MEAS_REFLECTION) {
772 try {
773 if (using_ext_gen) {
774 ext_gen->rf_off();
775 } else {
776 vna->rf_off(port_num);
777 }
778
779 logger::log(LEVEL_TRACE, "Port {} disabled", port_num);
780 } catch (int error_code) {
781 logger::log(LEVEL_ERROR, "Can't disable port {}", port_num);
782 return acquired_data;
783 }
784 }
785 }
786
787 acquired_data.insert_angles(get_current_angles());
788
789 if (using_ext_gen) {
790 acquired_data.insert_freq(get_current_freq());
791 } else {
792 acquired_data.insert_freq_list(get_freq_list());
793 }
794
795 logger::log(LEVEL_DEBUG, "Data acquired");
796
797 return acquired_data;
798}
