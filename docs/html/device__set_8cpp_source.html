285data_t DeviceSet::get_data(std::vector<int> port_list) {
286 if (stop_requested) {
287 logger::log(LEVEL_WARN, "Device set stops measuring");
288 stop_requested = false;
289
290 return data_t{};
291 }
292
293 logger::log(LEVEL_TRACE, "Preparing to acquire data");
294 data_t acquired_data{};
295
296 try {
297 if (!traces_configured) {
298 vna->create_traces(port_list, using_ext_gen);
299 traces_configured = true;
300 }
301
302 logger::log(LEVEL_TRACE, "Traces created");
303 } catch (int error_code) {
304 logger::log(LEVEL_ERROR, "Can't create traces");
305 return acquired_data;
306 }
307
308 for (int port_pos = 0; port_pos < port_list.size(); ++port_pos) {
309 if (stop_requested) {
310 logger::log(LEVEL_WARN, "Device set stops measuring");
311 stop_requested = false;
312
313 return data_t{};
314 }
315
316 int port_num = port_list[port_pos];
317 logger::log(LEVEL_TRACE, "Port = {}", port_num);
318
319 if (meas_type == MEAS_TRANSITION && port_pos == 0) {
320 try {
321 if (using_ext_gen) {
322 ext_gen->rf_on();
323 } else {
324 vna->rf_on(vna->get_source_port());
325 }
326
327 logger::log(LEVEL_TRACE, "Source port enabled");
328 } catch (int error_code) {
329 logger::log(LEVEL_ERROR, "Can't enable source port");
330 return acquired_data;
331 }
332 } else if (meas_type == MEAS_REFLECTION) {
333 try {
334 if (using_ext_gen) {
335 ext_gen->rf_on();
336 } else {
337 vna->rf_on(port_num);
338 }
339
340 logger::log(LEVEL_TRACE, "Port {} enabled", port_num);
341 } catch (int error_code) {
342 logger::log(LEVEL_ERROR, "Can't enable port {}", port_num);
343 return acquired_data;
344 }
345 }
346
347 if ((port_pos == 0 && meas_type == MEAS_TRANSITION) || meas_type == MEAS_REFLECTION) {
348 try {
349 vna->trigger();
350 vna->init();
351
352 logger::log(LEVEL_TRACE, "Measurements restarted");
353 } catch (int error_code) {
354 logger::log(LEVEL_ERROR, "Can't restart measurement");
355 return acquired_data;
356 }
357 }
358
359 try {
360 acquired_data.insert_iq_port_data(vna->get_data(port_pos));
361 logger::log(LEVEL_TRACE, "Data for port {} acquired", port_num);
362 } catch (int error_code) {
363 logger::log(LEVEL_ERROR, "Can't acquire data for port {} from VNA", port_num);
364 return acquired_data;
365 }
366
367 if (meas_type == MEAS_TRANSITION && port_pos == port_list.size() - 1) {
368 try {
369 if (using_ext_gen) {
370 ext_gen->rf_off();
371 } else {
372 vna->rf_off(vna->get_source_port());
373 }
374
375 logger::log(LEVEL_TRACE, "Source port disabled");
376 } catch (int error_code) {
377 logger::log(LEVEL_ERROR, "Can't disable source port");
378 return acquired_data;
379 }
380 } else if (meas_type == MEAS_REFLECTION) {
381 try {
382 if (using_ext_gen) {
383 ext_gen->rf_off();
384 } else {
385 vna->rf_off(port_num);
386 }
387
388 logger::log(LEVEL_TRACE, "Port {} disabled", port_num);
389 } catch (int error_code) {
390 logger::log(LEVEL_ERROR, "Can't disable port {}", port_num);
391 return acquired_data;
392 }
393 }
394 }
395
396 acquired_data.insert_angles(get_current_angles());
397
398 if (using_ext_gen) {
399 acquired_data.insert_freq(get_current_freq());
400 } else {
401 acquired_data.insert_freq_list(get_freq_list());
402 }
403
404 logger::log(LEVEL_DEBUG, "Data acquired");
405
406 return acquired_data;
407}
