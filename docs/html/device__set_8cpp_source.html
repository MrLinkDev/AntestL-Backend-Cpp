661data_t DeviceSet::get_data(std::vector<int> port_list) {
662 if (stop_requested) {
663 logger::log(LEVEL_WARN, "Device set stops measuring");
664 stop_requested = false;
665
666 return data_t{};
667 }
668
669 logger::log(LEVEL_TRACE, "Preparing to acquire data");
670 data_t acquired_data{};
671
672 try {
673 if (!traces_configured) {
674 vna->create_traces(port_list, using_ext_gen);
675 traces_configured = true;
676 }
677
678 logger::log(LEVEL_TRACE, "Traces created");
679 } catch (int error_code) {
680 logger::log(LEVEL_ERROR, "Can't create traces");
681 return acquired_data;
682 }
683
684 for (int port_pos = 0; port_pos < port_list.size(); ++port_pos) {
685 if (stop_requested) {
686 logger::log(LEVEL_WARN, "Device set stops measuring");
687 stop_requested = false;
688
689 return data_t{};
690 }
691
692 int port_num = port_list[port_pos];
693 logger::log(LEVEL_TRACE, "Port = {}", port_num);
694
695 if (meas_type == MEAS_TRANSITION && port_pos == 0) {
696 try {
697 if (using_ext_gen) {
698 ext_gen->rf_on();
699 } else {
700 vna->rf_on(vna->get_source_port());
701 }
702
703 logger::log(LEVEL_TRACE, "Source port enabled");
704 } catch (int error_code) {
705 logger::log(LEVEL_ERROR, "Can't enable source port");
706 return acquired_data;
707 }
708 } else if (meas_type == MEAS_REFLECTION) {
709 try {
710 if (using_ext_gen) {
711 ext_gen->rf_on();
712 } else {
713 vna->rf_on(port_num);
714 }
715
716 logger::log(LEVEL_TRACE, "Port {} enabled", port_num);
717 } catch (int error_code) {
718 logger::log(LEVEL_ERROR, "Can't enable port {}", port_num);
719 return acquired_data;
720 }
721 }
722
723 if ((port_pos == 0 && meas_type == MEAS_TRANSITION) || meas_type == MEAS_REFLECTION) {
724 try {
725 vna->trigger();
726 vna->init();
727
728 logger::log(LEVEL_TRACE, "Measurements restarted");
729 } catch (int error_code) {
730 logger::log(LEVEL_ERROR, "Can't restart measurement");
731 return acquired_data;
732 }
733 }
734
735 try {
736 acquired_data.insert_iq_port_data(vna->get_data(port_pos));
737 logger::log(LEVEL_TRACE, "Data for port {} acquired", port_num);
738 } catch (int error_code) {
739 logger::log(LEVEL_ERROR, "Can't acquire data for port {} from VNA", port_num);
740 return acquired_data;
741 }
742
743 if (meas_type == MEAS_TRANSITION && port_pos == port_list.size() - 1) {
744 try {
745 if (using_ext_gen) {
746 ext_gen->rf_off();
747 } else {
748 vna->rf_off(vna->get_source_port());
749 }
750
751 logger::log(LEVEL_TRACE, "Source port disabled");
752 } catch (int error_code) {
753 logger::log(LEVEL_ERROR, "Can't disable source port");
754 return acquired_data;
755 }
756 } else if (meas_type == MEAS_REFLECTION) {
757 try {
758 if (using_ext_gen) {
759 ext_gen->rf_off();
760 } else {
761 vna->rf_off(port_num);
762 }
763
764 logger::log(LEVEL_TRACE, "Port {} disabled", port_num);
765 } catch (int error_code) {
766 logger::log(LEVEL_ERROR, "Can't disable port {}", port_num);
767 return acquired_data;
768 }
769 }
770 }
771
772 acquired_data.insert_angles(get_current_angles());
773
774 if (using_ext_gen) {
775 acquired_data.insert_freq(get_current_freq());
776 } else {
777 acquired_data.insert_freq_list(get_freq_list());
778 }
779
780 logger::log(LEVEL_DEBUG, "Data acquired");
781
782 return acquired_data;
783}
