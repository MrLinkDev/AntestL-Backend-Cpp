139int SocketServer::wait_client() {
140 if (listen(server, 0) == SOCKET_ERROR) {
141 logger::log(LEVEL_ERROR, "{} ({}): Can't start listening", tag, port);
142 return SOCKET_NOT_LISTENING;
143 }
144
145 logger::log(LEVEL_INFO, "{} ({}): Listening...", tag, port);
146
147 int client_address_size = sizeof(client_address);
148 if ((client = accept(server, (SOCKADDR*)&client_address, &client_address_size)) != INVALID_SOCKET) {
149 logger::log(LEVEL_INFO, "{} ({}): Connected client with address {}", tag, port, inet_ntoa(client_address.sin_addr));
150 return CLIENT_CONNECTED;
151 }
152
153 logger::log(LEVEL_ERROR, "{} ({}): No clients...", tag, port);
154 return CLIENT_NONE;

166std::string SocketServer::read_data() {
167 std::string data{};
168
169 char buffer[DEFAULT_BUFFER_SIZE] = { 0 };
170 bool finish = false;
171
172 while (true) {
173 if (recv(client, buffer, sizeof(buffer), 0) == SOCKET_ERROR) {
174 return std::string{};
175 }
176
177 for (int pos = 0; pos < DEFAULT_BUFFER_SIZE; ++pos) {
178 if (buffer[pos] == termination[0]) {
179 finish = true;
180 } else if (buffer[pos] == termination[1] && finish) {
181 memset(buffer, 0, sizeof(buffer));
182
183 logger::log(LEVEL_TRACE, "{} ({}): Got data from client = {}", tag, port, data);
184
185 return data;
186 } else if (buffer[pos] == '\0') {
187 continue;
188 } else {
189 if (finish) {
190 finish = false;
191 }
192
193 data.push_back(buffer[pos]);
194 }
195 }
196
197 memset(buffer, 0, sizeof(buffer));
198 }
199}
